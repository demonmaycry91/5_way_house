# Flask 專案開發
my-flask-app/
├── .env                                # (手動建立) 存放環境變數，如 SECRET_KEY 和資料庫連線資訊
├── .gitignore                          # Git忽略清單，確保虛擬環境、快取和敏感檔案不會被提交
├── LICENSE                             # 專案的 MIT 授權條款
├── README.md                           # 專案的整體說明文件
├── requirements.in                     # 開發時定義的主要 Python 套件依賴
├── requirements.txt                    # 由 pip-compile 產生的、鎖定所有依賴套件版本的詳細清單
├── run.py                              # 啟動整個 Flask 應用程式的進入點
├── structure.txt                       # (輔助檔案) 用於定義專案文件夾結構的文字檔
├── create_project.py                   # (輔助檔案) 根據 structure.txt 自動建立專案結構的腳本
|
├── instance/                           # 存放本地設定和資料庫檔案，此目錄不應被 Git 追蹤
│   ├── app.db                          # SQLite 資料庫檔案
│   ├── client_secret.json              # (手動加入) Google API 的憑證檔案
│   └── token.json                      # (自動產生) 使用者授權後儲存的 Google API token
|
└── my_app/                             # 應用程式的核心程式碼
    ├── __init__.py                     # 初始化 Flask 應用的工廠函式，並註冊所有藍圖與擴充套件
    ├── auth_commands.py                # 定義使用者驗證相關的命令列指令 (例如: flask auth create-user)
    ├── forms.py                        # 存放所有 Flask-WTF 表單類別，用於處理表單驗證和 CSRF 保護
    ├── models.py                       # 定義所有資料庫模型 (User, Location, BusinessDay, Transaction)
    │
    ├── routes/                         # 存放所有功能的路由 (Blueprints)
    │   ├── __init__.py                 # 將 routes 文件夾標示為一個 Python 套件
    │   ├── admin_routes.py             # 後台管理路由，用於 CRUD 操作營業據點
    │   ├── cashier_routes.py           # 收銀系統的所有核心路由 (儀表板, POS, 日結等)
    │   ├── google_routes.py            # 處理 Google 帳號授權與回呼的路由
    │   ├── main_routes.py              # 應用程式的主要路由 (例如首頁)
    │   └── ocr_routes.py               # OCR 功能的路由 (目前為預留)
    │
    ├── services/                       # 存放與外部服務溝通的商業邏輯
    │   └── google_service.py           # 封裝與 Google Drive 及 Google Sheets API 互動的邏輯
    │   └── ocr_service.py              # (預留) 處理光學字元辨識 (OCR) 的核心邏輯 
    │
    ├── static/                         # 存放靜態檔案 (CSS, JavaScript, 圖片)
    │   ├── css/
    │   │   └── main.css                # (預留) 專案主要的自訂樣式表
    │   │   └── print.css               # 列印 PDF 報表時專用的樣式表
    │   ├── images/
    │   │   └── logo.png                # 應用程式 Logo
    │   └── js/
    │       └── pos_scripts.js          # POS 機前端互動的 JavaScript
    │       └── ocr_scripts.js          # (預留) OCR 功能前端互動的 JavaScript
    │
    └── templates/                      # 存放 HTML 樣板
        ├── admin/                      # 後台管理介面的樣板
        │   ├── location_form.html      # 新增與編輯據點的表單頁面
        │   └── locations.html          # 顯示所有據點的列表頁面
        ├── cashier/                    # 收銀系統相關的 HTML 頁面
        │   ├── close_day_form.html     # 日結作業 (現金盤點) 的表單頁面
        │   ├── daily_report.html       # 顯示每日報表的頁面
        │   ├── dashboard.html          # 使用者登入後看到的營運儀表板
        │   ├── login.html              # 使用者登入頁面
        │   ├── pos.html                # 銷售點終端 (POS) 操作介面
        │   ├── report_print.html       # 產生 PDF 報表專用的 HTML 樣板
        │   ├── settings.html           # 系統設定頁面 (如連結 Google 帳號)
        │   └── start_day_form.html     # 開店作業的表單頁面
        ├── ocr/                        # (預留) OCR 功能相關的樣板 
        │   └── scan_form.html          # (預留) OCR 功能相關的樣板 
        ├── base.html                   # 所有頁面的基礎樣板，包含全局導覽列和版面容器
        ├── index.html                  # 應用程式首頁
        └── macros.html                 # 存放可重複使用的 Jinja2 宏，如標準化的頁首
